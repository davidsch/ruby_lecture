## Most Things Changeable at Runtime ##
### Example: Vector, Operator Definitions ###


!SLIDE
## Reflection ##
    @@@ ruby
    class Foo
      def self.hi()
        puts "hi"
      end
      def self.bye()
        puts "bye"
      end
    end

    Foo.singleton_methods()
    => [:hi, :bye]


!SLIDE
## "Late Bound Everything" ##
    @@@ ruby
    def b(i)
      a(i)
    end

    def a(i)
      a * 2
    end

    >> b(10)
    => 20

    def a(i)
      a/2
    end

    >> b(10)
    => 5
