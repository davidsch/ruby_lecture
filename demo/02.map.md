!SLIDE
    @@@ ruby
    class Array
      def my_map()
        raise ArgumentError unless block_given?
        i = 0
        a = []
        while i < self.length
          a << yield(self[i])
          i += 1
        end
        a
      end
    end

    class Array
      def my_map()
        raise ArgumentError unless block_given?
        Array.new(self.length) { |i|
          yield self[i]
        }
      end
    end
